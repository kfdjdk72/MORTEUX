local function safeSetClipboard(text)
    local ok = pcall(setclipboard, text)
    if not ok then pcall(function() clipboard.set(text) end) end
end

local function createParticles(parent, count, color)
    if not parent then return end
    for _ = 1, count do
        local p = Instance.new("Frame")
        p.Size = UDim2.new(0, math.random(3,8), 0, math.random(3,8))
        p.Position = UDim2.new(math.random()*0.8+0.1, 0, math.random()*0.8+0.1, 0)
        p.BackgroundColor3 = color
        p.BorderSizePixel = 0
        p.ZIndex = 9999998
        p.Parent = parent
        local c = Instance.new("UICorner"); c.CornerRadius = UDim.new(1,0); c.Parent = p
        local t = TweenService:Create(p, TweenInfo.new(math.random(10,25)/10, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Position = UDim2.new(math.random()*0.9, 0, math.random()*0.9, 0),
            BackgroundTransparency = 1,
            Size = UDim2.new(0, math.random(1,3), 0, math.random(1,3)),
            Rotation = math.random(-360,360)
        })
        t:Play()
        t.Completed:Connect(function() if p then p:Destroy() end end)
    end
end

local function addStarBackground(parent)
    for _ = 1, 40 do
        local star = Instance.new("Frame")
        local sz = math.random(1,3)
        star.Size = UDim2.new(0, sz, 0, sz)
        star.Position = UDim2.new(math.random(), 0, math.random(), 0)
        star.BackgroundColor3 = Color3.fromRGB(255, math.random(100,200), math.random(100,150))
        star.BackgroundTransparency = math.random(3,8)/10
        star.BorderSizePixel = 0
        star.ZIndex = 999990
        star.Parent = parent
        local corner = Instance.new("UICorner"); corner.CornerRadius = UDim.new(1,0); corner.Parent = star
        task.delay(math.random(0,30)/10, function()
            if not star.Parent then return end

end

local function playOpenAnim(frame, targetW, targetH, centerY)
    frame.Visible = false
    frame.ClipsDescendants = true
    frame.Size = UDim2.new(0, targetW, 0, targetH)
    frame.Position = UDim2.new(0.5, 0, -0.6, 0)
    frame.BackgroundTransparency = 0.25

    for _, d in ipairs(frame:GetDescendants()) do
        if d:IsA("TextLabel") or d:IsA("TextButton") then
            d.TextTransparency = 1
            d.TextStrokeTransparency = 1
        end
    end

    task.wait()
    task.wait()
    frame.Visible = true

    local anim = TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
        Position = UDim2.new(0.5, 0, centerY, 0)
    })
    anim:Play()

    anim.Completed:Connect(function()
        frame.ClipsDescendants = false
    end)

    task.delay(0.25, function()
        for _, d in ipairs(frame:GetDescendants()) do
            if (d:IsA("TextLabel") or d:IsA("TextButton")) and d.Parent then
                TweenService:Create(d, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
                    TextTransparency = 0,
                    TextStrokeTransparency = 0.05
                }):Play()
            end
        end
    end)
end

local function openLoader()
    LoaderFrame.Visible = false
    LoaderFrame.ClipsDescendants = true
    LoaderFrame.Size = UDim2.new(0, 700, 0, 250)
    LoaderFrame.Position = UDim2.new(0.5, 0, -0.6, 0)
    LoaderFrame.BackgroundTransparency = 0.25

    for _, d in ipairs(LoaderFrame:GetDescendants()) do
        if d:IsA("TextLabel") or d:IsA("TextButton") then
            d.TextTransparency = 1
            d.TextStrokeTransparency = 1
        end
    end

    task.wait()
    LoaderFrame.Visible = true

    local anim = TweenService:Create(LoaderFrame, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
        Position = UDim2.new(0.5, 0, 0.5, 0)
    })
    anim:Play()
    anim.Completed:Connect(function()
        LoaderFrame.ClipsDescendants = false
    end)

    task.delay(0.25, function()
        for _, d in ipairs(LoaderFrame:GetDescendants()) do
            if (d:IsA("TextLabel") or d:IsA("TextButton")) and d.Parent then
                TweenService:Create(d, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
                    TextTransparency = 0, TextStrokeTransparency = 0.05
                }):Play()
            end
        end
    end)
end
