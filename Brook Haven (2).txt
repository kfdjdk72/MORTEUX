local function CreateButton(name, callback)
    local state = false
    local Btn = Instance.new("TextButton")
    local BtnCorner = Instance.new("UICorner")
    
    Btn.Name = name
    Btn.Parent = TabContainer
    Btn.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
    Btn.Size = UDim2.new(1, 0, 0, 38)
    Btn.Font = Enum.Font.GothamMedium
    Btn.Text = name .. ": OFF"
    Btn.TextColor3 = Color3.fromRGB(180, 180, 180)
    Btn.TextSize = 13.000
    
    BtnCorner.CornerRadius = UDim.new(0, 6)
    BtnCorner.Parent = Btn
    
    Btn.MouseButton1Click:Connect(function()
        state = not state
        Btn.TextColor3 = state and Color3.fromRGB(255, 0, 0) or Color3.fromRGB(180, 180, 180)
        Btn.Text = name .. ": " .. (state and "ON" or "OFF")
        callback(state)
    end)
end

SpeedBox.TextSize = 13.000

SpeedBoxCorner.Parent = SpeedBox

SpeedBox.FocusLost:Connect(function(enter)
    if enter then

end)

local JumpBox = Instance.new("TextBox")
local JumpBoxCorner = Instance.new("UICorner")

JumpBox.Parent = TabContainer
JumpBox.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
JumpBox.Size = UDim2.new(1, 0, 0, 38)
JumpBox.Font = Enum.Font.GothamMedium
JumpBox.PlaceholderText = "Enter Jump Power"
JumpBox.Text = ""
JumpBox.TextColor3 = Color3.fromRGB(255, 255, 255)
JumpBox.TextSize = 13.000
JumpBoxCorner.CornerRadius = UDim.new(0, 6)
JumpBoxCorner.Parent = JumpBox

JumpBox.FocusLost:Connect(function(enter)
    if enter then
        local v = tonumber(JumpBox.Text)
        if v then 
            game.Players.LocalPlayer.Character.Humanoid.UseJumpPower = true
            game.Players.LocalPlayer.Character.Humanoid.JumpPower = v 
        end
    end
end)

CreateButton("Infinite Jump", function(v)
    getgenv().InfJump = v
    game:GetService("UserInputService").JumpRequest:Connect(function()
        if getgenv().InfJump then
            game.Players.LocalPlayer.Character:FindFirstChildOfClass("Humanoid"):ChangeState("Jumping")
        end
    end)
end)

CreateButton("No Clip", function(v)
    getgenv().noclip = v
    game:GetService("RunService").Stepped:Connect(function()
        if getgenv().noclip then
            for _, v in pairs(game.Players.LocalPlayer.Character:GetDescendants()) do
                if v:IsA("BasePart") then v.CanCollide = false end
            end
        end
    end)
end)

CreateButton("Teleport Bank", function(v)
    if v then
        game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(-400, 23, -200)
    end
end)
